import { useCallback, useEffect } from "react";
import useEmblaCarousel from "embla-carousel-react";

const IMAGES = [
  // Prefer the original list used previously.
  // Fallback: enumerate files under /assets/dovecote/gallery/
  // Example placeholders – replace with actual filenames in that folder:
  "/assets/dovecote/gallery/aerial-01.jpg",
  "/assets/dovecote/gallery/aerial-02.jpg",
  "/assets/dovecote/gallery/masterplan-01.jpg",
  "/assets/dovecote/gallery/coastline-01.jpg",
];

export default function ProjectGallery() {
  const [emblaRef, emblaApi] = useEmblaCarousel({ loop: true });
  useEffect(() => {
    if (!emblaApi) return;
    let t: any;
    const play = () => { t = setInterval(() => emblaApi.scrollNext(), 5000); };
    const stop = () => { if (t) clearInterval(t); };
    play();
    emblaApi.on("pointerDown", stop);
    emblaApi.on("pointerUp", play);
    return () => stop();
  }, [emblaApi]);

  const scrollTo = useCallback((idx: number) => emblaApi?.scrollTo(idx), [emblaApi]);

  const selectedIndex = emblaApi?.selectedScrollSnap() ?? 0;
  const snaps = emblaApi?.scrollSnapList() ?? IMAGES.map((_, i) => i);

  return (
    <section id="project-gallery" className="py-16 md:py-24 bg-white">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <header className="max-w-3xl">
          <h2 className="text-3xl md:text-4xl font-heading font-bold">Project Gallery</h2>
          <p className="mt-3 text-gray-600 text-base md:text-lg">
            Explore aerial views and master planning of Dovecote Estate Stanley&apos;s exclusive coastal subdivision.
          </p>
        </header>

        <div className="mt-8">
          <div className="overflow-hidden rounded-2xl shadow" ref={emblaRef}>
            <div className="flex">
              {IMAGES.map((src, i) => (
                <div className="min-w-0 flex-[0_0_100%]" key={src}>
                  <div className="aspect-[16/9] md:aspect-[16/9] w-full overflow-hidden bg-slate-100">
                    <img
                      src={src}
                      alt={
                        i === 0
                          ? "Aerial of Dovecote Estate and coastline"
                          : "Dovecote Estate project gallery image"
                      }
                      className="h-full w-full object-cover"
                      loading={i === 0 ? "eager" : "lazy"}
                    />
                  </div>
                </div>
              ))}
            </div>
          </div>

          {/* Dots */}
          <div className="mt-4 flex justify-center gap-2">
            {snaps.map((_, i) => (
              <button
                key={i}
                aria-label={`Go to slide ${i + 1}`}
                className={`h-2.5 w-2.5 rounded-full transition ${
                  i === selectedIndex ? "bg-gray-900" : "bg-gray-300 hover:bg-gray-400"
                }`}
                onClick={() => scrollTo(i)}
              />
            ))}
          </div>
        </div>
      </div>
    </section>
  );
}
